    1.2.构建编码器与解码器：¶

    stoi = { ch:i for i,ch in enumerate(chars) }
itos = { i:ch for i,ch in enumerate(chars) }
encode = lambda s: [stoi[c] for c in s] 
decode = lambda l: ''.join([itos[i] for i in l]) 

print(encode("hi I'am here"))
print(decode([46, 47, 1, 21, 5, 39, 51, 1, 46, 43, 56, 43]))

    运行结果：
    [46, 47, 1, 21, 5, 39, 51, 1, 46, 43, 56, 43]
    hi I'am here
    这里采用的是最简单的按照字符位置编码，当然个人觉得也可以再加特殊字符以表示句子的开始结束

    训练块大小定义成8，意思是1推2，1和2推3，以此类推。训练块和当前大模型所说的上下文长度是一回事。个人更喜欢理解为滑动窗口训练 

    def get_batch(split):
    # generate a small batch of data of inputs x and targets y
    data = train_data if split == 'train' else val_data
    ix = torch.randint(len(data) - block_size, (batch_size,))
    x = torch.stack([data[i:i+block_size] for i in ix])
    y = torch.stack([data[i+1:i+block_size+1] for i in ix])
    return x, y

    定义了一个随机选取（batch，block）大小的数据

    2.模型构建¶

    2.1最简单的二元模型，损失，生成的构建：¶

    import torch
import torch.nn as nn
from torch.nn import functional as F
torch.manual_seed(1337)

class BigramLanguageModel(nn.Module):

    def __init__(self, vocab_size):
        super().__init__()
        self.token_embedding_table = nn.Embedding(vocab_size, vocab_size)

    def forward(self, idx, targets=None):
        #idx和target就可以理解是上面的xb和yb，二者的行状都是（B，T）
        logits = self.token_embedding_table(idx) # (B,T,C)

        if targets is None:
            loss = None
        else:
            B, T, C = logits.shape
            logits = logits.view(B*T, C)
            targets = targets.view(B*T)
            loss = F.cross_entropy(logits, targets)

        return logits, loss

    def generate(self, idx, max_new_tokens):
        for _ in range(max_new_tokens):
            logits, loss = self(idx)
            # 只关注最后一个
            logits = logits[:, -1, :] # becomes (B, C)
            # 通过softmax
            probs = F.softmax(logits, dim=-1) # (B, C)

            idx_next = torch.multinomial(probs, num_samples=1) # (B, 1)

            idx = torch.cat((idx, idx_next), dim=1) # (B, T+1)
        return idx

m = BigramLanguageModel(vocab_size)
logits, loss = m(xb, yb)
print(logits.shape)
print(loss)

print(decode(m.generate(idx = torch.zeros((1, 1), dtype=torch.long), max_new_tokens=100)[0].tolist()))

    从简单的二元模型开始理解，forward函数就是将idx通过self.token_embedding_table函数映射到了词表维度上，然后根据cross_entropy（）对传入参数的行状要求传入数据返回损失函数 generate（）函数就是刚刚理解的滑动窗口将输入的idx依次从词表概率中通过torch.multinomial函数抽出下一个单词 生成结果很随机，这次的运行结果是：Sr?qP-QWktXoL&jLDJgOLVz'RIoDqHdhsV&vLLxatjscMpwLERSPyao.qfzs$Ys$zF-w,;eEkzxjgCKFChs!iWW.ObzDnxA Ms$3

    2.2构建特征¶

    way1：

    xbow = torch.zeros((B,T,C))
for b in range(B):
    for t in range(T):
        xprev = x[b,:t+1] # (t,C)
        xbow[b,t] = torch.mean(xprev, 0)

    显而易见就是用之前量的平均值来实现对之前特征的提取 

    wei = torch.tril(torch.ones(T, T))
wei = wei / wei.sum(1, keepdim=True)
xbow2 = wei @ x # (B, T, T) @ (B, T, C) ----> (B, T, C)
torch.allclose(xbow, xbow2)

    这是通过向量计算快速得到的平均值特征

    way2(重头戏)：

    torch.manual_seed(1337)
B,T,C = 4,8,32 # batch, time, channels
x = torch.randn(B,T,C)

head_size = 16
key = nn.Linear(C, head_size, bias=False)
query = nn.Linear(C, head_size, bias=False)
value = nn.Linear(C, head_size, bias=False)
k = key(x)   # (B, T, 16)
q = query(x) # (B, T, 16)
wei =  q @ k.transpose(-2, -1) # (B, T, 16) @ (B, 16, T) ---> (B, T, T)

tril = torch.tril(torch.ones(T, T))
#wei = torch.zeros((T,T))
wei = wei.masked_fill(tril == 0, float('-inf'))
wei = F.softmax(wei, dim=-1)

v = value(x)
